Megoldás:

﻿using System;
using System.Collections;
using System.Collections.Generic;

namespace ConsoleApp75
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.Write("Adon meg zárójeleket tartalmazó kifejezést:");
            string zarojelek = Console.ReadLine();//"((3+2)/5)+(4-3)";
            int[] szinek = new int[zarojelek.Length];
            int szin = 4;
            Stack <int> verem= new Stack <int>();
            bool hiba = false;
            for (int i = 0; i < zarojelek.Length; i++)
            {
                if (zarojelek[i] == '(')
                {
                    verem.Push(i);
                }
                else if (zarojelek[i] == ')')
                {
                    if (verem.TryPop(out int j))
                    {
                        szinek[j] = szin;
                        szinek[i] = szin++;
                    }
                    else
                    {
                        Console.WriteLine("Hibás zárójelezés! (párnélküli bezáró zárójel van)");
                        hiba = true;
                    }
                }
                else { 
                    szinek[i] = 7; 
                }
            }
            if (verem.Count != 0)
            {
                Console.WriteLine("Hibás zárójelezés! (pár nélküli nyitó zárójel van)");
                hiba = true;
            }
            if(!hiba)
                for (int i = 0; i < zarojelek.Length; i++)
                {
                    Console.ForegroundColor = (ConsoleColor)szinek[i];
                    Console.Write(zarojelek[i]);
                }
            Console.ReadKey();
        }
    }
}
